-- تكدر تخلي شخص يشغل سكربت وشخص ميكدر يشغله

local correctUserId = 7094200192  -- هنا حط ايدي 

local screenGui = Instance.new("ScreenGui")
screenGui.Name = "IDCheckGUI"
screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

local textLabel = Instance.new("TextLabel")
textLabel.Size = UDim2.new(0, 400, 0, 100)
textLabel.Position = UDim2.new(0.5, -200, 0.5, -50)
textLabel.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
textLabel.BackgroundTransparency = 0.3
textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
textLabel.Font = Enum.Font.GothamBold
textLabel.TextSize = 20
textLabel.TextScaled = false
textLabel.TextWrapped = true
textLabel.BorderSizePixel = 2
textLabel.BorderColor3 = Color3.fromRGB(255, 255, 255)
textLabel.Parent = screenGui

local player = game.Players.LocalPlayer

if player.UserId == correctUserId then

    textLabel.Text = " ايديك صح!\nUser ID: " .. player.UserId .. "\nتم تشغيل!"
    textLabel.BorderColor3 = Color3.fromRGB(0, 255, 0)
    textLabel.BackgroundColor3 = Color3.fromRGB(0, 100, 0)
    
    wait(3)
    textLabel:Destroy()
    
    print("سكربت شغال")
    
else

    textLabel.Text = "ايدي غلط!\nUser ID الحالي: " .. player.UserId .. "\nالمطلوب: " .. correctUserId .. "\nفشل تشغيل"
    textLabel.BorderColor3 = Color3.fromRGB(255, 0, 0)
    textLabel.BackgroundColor3 = Color3.fromRGB(100, 0, 0)
    
    wait(5)
    game:Shutdown()
end

if player.UserId == correctUserId then
    wait(10)
    if screenGui and screenGui.Parent then
        screenGui:Destroy()
    end
end
